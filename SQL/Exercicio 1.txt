	CODIGO INT PRIMARY KEY,
	NOME VARCHAR(100),
	PRECO DECIMAL(9,2),
	PESO DECIMAL(9,2)
);

SELECT * FROM PRODUTO
ORDER BY PRECO ASC;

SELECT * FROM CLIENTE
ORDER BY NOME DESC;

SELECT SUM(PRECO) AS TOTAL_PRECOS
FROM PRODUTO;

SELECT CIDADE, COUNT(*) AS TOTAL_CLIENTES
FROM CLIENTE GROUP BY CIDADE;
